	2.	Nettoyage
		Si necessaire:
			chmod +x clean
			sudo apt install python3 python-is-python3
		./clean countryInfo.txt > country-clean.txt
		./clean canada/CA.txt > CA-clean.txt
	3. Supprimer commentaires
		grep -v ^# country-clean.txt > country-nocomment.txt
